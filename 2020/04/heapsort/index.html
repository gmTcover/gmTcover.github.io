void heapify(int tree[],int n,int i){
	if(i >= n){
		return ;
	}
	int c1 = 2*i +1;
	int c2 = 2*i + 2;
	int max = i;
	if(c1 < n && tree[c1] > tree[max]){
		max = c1;
	}
	if(c2 < n && tree[c2] > tree[max]){
		max = c2;
	}
	if(max != i){//è¿™é‡Œä¹Ÿéšè—äº†ä¸€ä¸ªé€’å½’ç»ˆæ­¢æ¡ä»¶

void build_heap(int tree[], int n){
	int last_node = n - 1;
	int parent = (last_node - 1) / 2;
	for(int i = parent; i >= 0; i--){
		heapify(tree,n,i);
	}
}

void heap_sort(int tree[], int n){
	build_heap(tree, n); 
	for(int i = n -1; i >= 0; i--){
		swap(tree, i, 0);
		heapify(tree,i, 0);
	}
}
