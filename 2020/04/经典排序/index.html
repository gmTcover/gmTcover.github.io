void bubble(int arr[], int n) {
    int tmp;
    for (int i = 0; i < n -1; i++) {
        if(arr[i] > arr[i+1]) {
            tmp = arr[i];
            arr[i] = arr[i+1];
            arr[i+1] = tmp;
        }
    }
}
void bubbleSort(int arr[], int n) {
    for (int i = n ; i > 1; i--) {
        bubble(arr,i);
    }
}
int findMaxIdx(int arr[], int n) {
    int maxIdx = 0, max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
            maxIdx = i;
        }
    }
    return maxIdx;
}
void selectSort(int arr[], int n) {
    int tmp;
    for (int i = n ; i > 1; i--) {
        int idx = findMaxIdx(arr,i);
        tmp = arr[i-1];
        arr[i-1] = arr[idx];
        arr[idx] = tmp;
    }
}

void insert(int arr[], int n) {
    int i = n-1, value = arr[i];
    while (arr[i-1] > value) {
        arr[i] = arr[i-1];
        i--;
        if(i == 0) break;
    }
    arr[i] = value;
}
void insertSort(int arr[], int n) {
    for(int i = 1; i < n; i++) {
        insert(arr,i+1);
    }
}
