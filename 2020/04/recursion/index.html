#include <iostream>
using namespace std;
int max(int* arr, int n) {
    if (n == 0) return arr[0];
    return max(arr, n-1) > arr[n]? max(arr, n-1):arr[n];
}
int dp_max(int* arr, int n) {
    int ret = arr[0];
    for (int i=1; i<n; i++) {
        if (arr[i] > ret) {
            ret = arr[i];
        }
    }
    return ret;
}
int re_sum(int n) {
    if (n == 1) return 1;
    return re_sum(n-1) + n;

}
int re_findMax(int* arr, int L, int R) {
    if (L==R) {
        return arr[L];
    } else {
        return (re_findMax(arr, L, R-1) > arr[R])? re_findMax(arr, L, R-1): arr[R];
    }
}
int re_findMax1(int arr[], int L, int R) {
    if (L==R) {
        return arr[L];
    } else {
        int a = arr[L];
        int b = re_findMax1(arr, L+1, R);
        return a>b ? a:b;
    }
}

int re_sum1(int arr[], int L, int R) {
    if (L==R) {
        return arr[L];
    } else {
        return re_sum1(arr, L+1, R) + arr[L];
        //return sum(arr,L, R-1) + arr[R];
    }
}
void bubble(int arr[], int L, int R) {
    if (L<R) {
        for (int i=L; i<R-1; i++) {
            if (arr[i] > arr[i+1]) {
                int temp = arr[i];
                arr[i] = arr[i+1];
                arr[i+1] = temp;
            }
        }
        bubble(arr, L, R-1);//heapfity
    }
}
int main() {
    int arr[] = {7, 4, 8, 6, 3, 14, 9, 11, 5};
    printf("max = %d\n", max(arr,8));
    printf("dp_max = %d\n", dp_max(arr,8));

    int sum = re_sum(10);
    printf("sum = %d\n", sum);

    printf("re_findMax = %d\n", re_findMax(arr, 6, 8));
    printf("re_findMax1 = %d\n", re_findMax1(arr, 6, 8));

    int arr1[] = {1,2,3,4,5,6};
    printf("sum = %d\n", re_sum1(arr1,0,5));
    return 0;
}
