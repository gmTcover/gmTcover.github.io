Class内部函数调用

十一月 5, 2019 发布在 c++基础, Test 
#include <iostream>
using namespace std;

class A 
{
	public:
		A(int ii):i(ii) { cout << "A::A()" << endl; }//这些都是inline;
		//A(){} 没有参数的话 子类B default constructor 编译器会自动生成，虽然A()什么都没做
		//因为编译器不能假定A的构造函数做了什么样的操作，因此有B,C必要生成B的构造函数，保证A类型的数据成员的构造函数正常执行
		//
		~A() { cout << "A::~A()" << endl; }

		void print() { cout << "A.a =" << i << endl; }
//protected:
		void set(int i) { this->i = i; }//一个修改i的借口
	private:
		int i;
};

class B : public A//B是 A的子类，继承
{
	public:
	B():A(15){ cout << "B::B()" << endl;}//B与A哪个先构造(Person,student模型，person在student里面)，父类A先构造然后B
	~B(){ cout << "B::~B()" << endl; }
};

class C //对象的组合
{
public:
	C() : a(20){ cout << "C::C()" << endl; } 
	~C() { cout << "C::~C()" << endl;};
	void print();//体现出了与类的继承区别，要构造新对象的函数
	void set(int i){ a.set(i);}
private:
	A a;//无法直接访问这个私有值，没有protected是类继承里面的
};


void C::print(){ a.print();}
int main()
{
	B b;
	b.print();
	//cout << b.i << endl;mian 函数里面无法访问，通过指针可以访问 int *p = （int*）&a，对 B来说  p++（sizeof A，sizeof B，判断寻找指针），即b.i，可以修改，不在乎private，所以c++都是假的，指针式万能的说法由来
	int *p = (int*)&b;
	//cout << b.i << endl;
	cout << "私有值" << *p << endl;
	b.set(10);
	b.print();

	cout << "C what" << endl;
	C cc;
	cc.print();
	cc.set(100);
	cc.print();

	return 0;
}
