#include<iostream>
using namespace std;
//std::string using namespace std 全名
class Employee 
{
	public:
		Employee( const std::string& name, const std::string& ssn );
		const std::string& get_name() const;
		void print( std::ostream& out ) const;
		void print( std::ostream& out, const std::string& msg ) const;

	protected:
		std::string m_name;//这样设计今后的子类可以直接访问进行修改,有好有坏，不知道子类今后会不会修改，设计上以后无论怎么继承，不能修改我初始的定义
		std::string m_ssn;
};

Employee::Employee( const string& name, const string& ssn ):m_name(name), m_ssn(ssn){
	//initilizer list sets up the values!
}

inline const std::string& Employee::get_name() const
{
	return m_name;
}

inline void Employee::print( std::ostream& out ) const//两个print函数，重载函数，参数不同，会自动调用，
{
	out << m_name << endl;
	out << m_ssn << endl;
}

inline void Employee::print( std::ostream& out, const std::string& msg) const
{
	out << msg << endl;
	print(out);//程序设计思路，尽量避免 code duplication ，便于修改
}

class Manager : public Employee
{
	public:
		Manager(const std::string& name, const std::string& ssn, const std::string& title );
		const std::string title_name() const;
		const std::string& get_title() const;
		void print(std::ostream& out) const;//有问题,与父类print重名overrided，解决方法？
	private:
		std::string m_title;
};

Manager::Manager( const string& name, const string& ssn, const string& title = "" ) 
	:Employee(name, ssn ), m_title( title ) { } 

inline void Manager::print( std::ostream& out ) const
{
	Employee::print( out ); //call the base class print 语法
	out << m_title << endl;

}

inline const std::string& Manager::get_title() const 
{
	return m_title;
}

inline const std::string Manager::title_name() const
{
	return string( m_title +":" + m_name );// access base m_name
}
