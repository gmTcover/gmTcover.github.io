cosnt Integer& Integer::operator++(){
	*this += 1;//increment
	return *this;//fetch
}

const Integer Integer::operator++(int){
	Integer old( *this );//fetch 传入对象发生拷贝构造 *this 是个对象
	++(*this);//increment 第一个++定义这个函数
	return old;//return 新对象，在发生一次拷贝构造，创建一块空间
}
}

Relational operators

  • 都是inlne

    bool Integer::operator==( const Integer& rhs ) const {
	return i == rhs.i;// this->i;*this == rhs.i 返回true or false
}


bool Integer::operator!=( const Integer& rhs ) const {
	return !(*this == rhs);//调用上一个，为什么要这么做
}//*this == rhs 为什么可以 i呢？ 因为里面首位置就是member i

bool Integer::operator<( cosnt Integer& rhs ) const {
	return i < rhs.i;
}

bool Integer::operator>( const Integer& rhs ) const {
	return rhs < *this;//思考一下巧妙之处

}
bool Integer::operator<=( const Integer& rhs) const {
	return !(rhs < *this );
}


bool Integer::operator>=( const Integer& rhs )const {
	return !(*this < rhs);
}

- 6个bool原函数，指定义了 ==，<,其他4个是这2个改造的，好处？
- 代码修改,只有2个函数接触到了i，其他都是用类的this指针表示
- 都是inline，没有性能损失

Operator[]

  • 类表达一种容器的概念，可以index来访问

  • must be a member function, single argument, so it should be return a reference

    Vtctor v(100)

v[10] = 45;

*v[10] = 45 ; //C ++ 写法

#ifndef _VECTOR_H_
#define _VECTOR_H_

class Vector {
	public:
	Vector(int size):m_size(size){
		m_array = new int[size];

	}
	virtual ~Vector(){ delete m_array; }

	int& operator[](int index){ return m_array[index];}//不是这几访问m_array自救就行为什么要多次一举，设计[]的运算符
private:
	int m_size;
	int *m_array;
};
#endif
