// c语言中有 FILE指针

 FILE* fp = fopen("file","r");一个文件名，“r”为读写
	if(fp){
	fscanf(fp,...);
	fclose(fp);
	} else {};


int main (int argc, char *argv[])
{
	FILE *fp = fopen("12.in","r");
	if( fp ){
		int num;
		fscanf(fp,"%d",&num);
		printf("%d\n",num);
		fclose(fp);
	} else {
		printf("文件名无法打开\n");
	}
	return 0;
}
// r :打开只读
// r+：打开读写，从文件头开始
// w：打开只写。如果不存在则新建，如果存在则清空
// w+：打开x先读(之前已经写入的内容)写。如果不存在则新建，如果存在则清空
// a：打开追加。如果不存在在则新建，如果存在则从文件尾开始
// ..x ：只新建，如果文件已存在则不能打开，w和a之后，如果文件已存在和不能打开，保证文件不被破坏wx；ax
// wt,文本方式写文件会（对应二进制模式读取时有可能换行符\n会进行转换\r\n，文件读取不完整）wb：二进制模式输入
