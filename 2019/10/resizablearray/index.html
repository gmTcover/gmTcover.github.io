可变数组

#include <stdio.h>
#include <stdlib.h>

const int BLOCK_SIZE = 20;//每次inflate大小
typedef struct{
	int *array;
	int size;
} Array;//不要定义处指针类型*Array，有缺点，不易阅读，且 Array a创建的是指针，本地函数内部不能创建同名的 Array a


Array array_creat(int init_size);
void array_free(Array *a);
int array_size(const Array *a);
int* array_at(Array *a,int index);
void array_inflate(Array *a,int more_size);

int main(int argc, char const *argv[])
{
	Array a = array_creat(100);
	//printf("%d\n",a.size);
	printf("%d\n",array_size(&a));//封装了a的 size，现在简单的程序a.size就够了，今后内部程序改进，函数把内部细节保护起来了
	*array_at(&a,0) = 10;//返回时指针有两个便利的地方，*p可以得到现在位置，也能通过*p 修改内容
	printf("%d\n",*array_at(&a,0));

	int number =0;
	int cnt = 0;
	while (number != -1){//怎么改进这个循环？
		sacnf("%d",&number);
		if( number != -1)
			*array_at(&a,cnt++) = number;
		//scanf("%d",array_at(&a,cnt++));
	}
	array_free(&a);//free array_creat函数malloc的内存
	return 0;
}

Array array_creat(int init_size)//返回一个结构体，灵活使用，一种方法， Array a = array_creat(size); 
{
	Array a;
	a.size = init_size;
	a.array = (int*)malloc(sizeof(int)*init_size);//申请了内存，但是没有free
	a.size = init_size;
	return a;
}

void array_free(Array *a)
{
	free(a->array);
	a->size = 0;
	a->array = NULL;
}

int array_size(const Array *a)//为什么要这么复杂，不直接a.size,为了封装，保护了内部实现细节，你不知到我怎么做的
{
	return a->size;
}

int* array_at(Array *a, int index)//封装
{
	if ( index >= a->size )//数组越界了，这里只考虑的大于0情况
	{
		array_inflate(a,(index/BLOCK_SIZE+1)*BLOCK_SIZE-a->size);//减a->size是为了 保证增加以后block都是一块一块固定大小
	}
	return &(a->array[index]);//定义了返回一个int*指针的好处，可以随时修改下标的值，也可以分解成2步，先返回一个int
}

void array_inflate(Array *a, int more_size)
{
	int *p = (int*)malloc(sizeof(int)*(a->size + more_size));
	int i;
	for ( i=0; i<a->size; i++ ){
		p[i] = a->array[i];//可以换成库含量里面的循环memcpy比for循环效率更高 这里实现的是复制以前的值到新的空间
	}
	free(a->array);
	a->array = p;
	a->size += more_size;
}
