如何做输出

十月 15, 2019 发布在 JavaTest, 多项式相加 
import java.util.Scanner;

public class polynomialADD {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner in = new Scanner(System.in);
		int idx = in.nextInt();
		int off = in.nextInt();
		boolean flag = false; //判断输出正负号 ，首位输出的套路
		int[] numOfidx = new int[101];
		int idx_max = 0;//优化减小输出的循环范围，本来要重小于101开始
		
		do//此处做何种循环
		{
		idx = in.nextInt();
		off = in.nextInt();
		numOfidx[idx] = off;
		if ( idx > idx_max)
		{
			idx_max = idx;
		}
		}while(idx !=0);
		do
		{
		idx = in.nextInt();
		off = in.nextInt();
		numOfidx[idx] += off;
		if ( idx > idx_max)
		{
			idx_max = idx;
		}
		}while(idx !=0);
		
		for( int i = idx_max; i >1; i--)
		{
			if(numOfidx[i] != 0)
			{
				if(flag && numOfidx[i] > 0) System.out.print("+");
				if(numOfidx[i] < 0 ) System.out.print("-");
				flag = true;// 首位不输出以后每次输出
				System.out.print(numOfidx[i]+"x"+i);
			}
			else
				continue;
		} 
		
		if(numOfidx[1] != 0)
		{
			if(numOfidx[1] > 0) System.out.print("+");
			System.out.print(numOfidx[1]+"x");
		}
			
		if(numOfidx[0] !=0)
		{
			if(numOfidx[0] > 0) System.out.print("+");
			System.out.print(numOfidx[0]);
		
		}
		if(!flag) System.out.print("0");
		in.close();
	}
}
