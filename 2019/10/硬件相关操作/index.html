MCU的SFR

  • 器件手册

  • 特殊功能寄存器

    const unsigned int SBS = 1u << 2;//stop bit select，100
const unsigned int PE = 1u << 3;//parity enable，1000
UOLCR |= SBS | PE;//UOLCR - 0xE000C00C，特殊寄存器 1100
UOLCR &= ~SBS;//取反... ... 011 ,共32位，使第三位置零
UOLCR &= ~PE;//... ... 01111

UOLCR &= ~(SBS | PE);//更改2个位置的bit 为0

位段

  • 把一个int的若干位组合成一个结构

    struct {
	unsigned int leading : 3;//表示这个成员占3 bit
	unsigned int FLAG1: 1;
	unsigned int FLAG2: 1;
	int trailing: 11; 放入一个总的int中，变成位段中的变量
}

  • 根据器件手册列表

    #include <stdio.h>
void prtBin(unsigned int number);
struct U0 {
	unsigned int leading :3;//定义过结构3bit
	unsigned int FLAG1:1;//1bit
	unsigned int FLAG2:1;//1bit
	int trailing:27;//位段，构建一个结构内存大小位一个int
};

int main(int arc, char const *argv[])
{
	struct U0 uu;
	uu.leading = 2;//3个bit位上代表的值，单片机里做开关的作用
	uu.FLAG1 = 0;
	uu.FLAG2 = 1;
	uu.trailing = 0;//初始化

	printf("sizeof(uu)=%lu\n",sizeof(uu));// 4个字节
	prtBin(*(int*)&uu);//底层比较常见做法，强制类型转换输出结构指针，指针包含二进制
}

void prtBin(unsigned int number)//用了unsigned意义就是不包括正负0～2^32-1
{
	unsigned mask = 1u<<31;
	for(; mask; mask >>=1){
		printf("%d",number & mask?1:0);
	}
	printf("\n");
}
