C语言素数表continue解决方案

十月 14, 2019 发布在 C程序设计, Test

解决continue ，封装函数判断逻辑

#include <stdio.h>
int isPrime(int x,int knownPrimes[], int numberOfkownPrimes);
int main(int argc, char const *argv[])
{

	const int number = 100;
	int prime[number];
	prime[0] = 2;
	int cnt = 1;
	int i = 3;
	while ( cnt < number )
	{
		if( isPrime( i,prime,cnt))//封装 函数，isPrimeh函数的返回结果与prime[]表达式相关
		{
			prime[cnt++] = i;//套路，写入primt[cnt],并且cnt++，指向下一位。
		}
		i++;
	}

	for( i = 0; i < number; i++)
	{
		printf("%d",prime[i]);
		!((i+1) % 5)? printf("\n"):printf("\t");//从第一个计数开始
	}
	return 0;
}
int isPrime(int x,int knownPrimes[], int numberOfkownPrimes)
{
	int ret = 1;
	for( int i=0; i < numberOfkownPrimes; i++)
	{
		if( x % knownPrimes[i] == 0)
		{
			ret = 0;
			break;
		}
	}
	return ret;
}
