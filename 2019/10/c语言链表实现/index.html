#include <stdio.h>
#include <stdlib.h>

typedef struct _node {
	int value;
	struct _node *next;
} Node;

typedef struct _list {
	Node* head;
    Node* tail;//保存最后位置，工程设计，不是仅仅是 放入东西，插入东西
}List;
void add(List* plist, int number);

			{
			q->next = p->next;//如何找到界限，narrow 看到指针标记，保证指针的安全等于NULL，不等于NULL的情况
			} else

			printf("删除后的结果：\n");

void add(List* plist, int number)
{
	// add to linkded-list
	Node *p = (Node*)malloc(sizeof(Node));
	p->value = number;
	p->next = NULL;
	Node *last = plist->head;//每次加入一个新的数据，都是重plist->head开始，大量数据是时间复杂度效率低下，修改：第一次加完收据是，记录最后数据：plist->tail = p
	if ( last){ 
		while ( last->next ){//指针从开始遍历到最后一个指向NULL跳出循环
			last = last->next;
		}
		last->next = p;//把NULL变成新申请的p指针
		
	} else {
		plist->head = p;	
	}
}

void print(List* plist)//传一个结构不如传指针效率
{
	Node *p;
	for ( p=plist->head; p; p= p->next){//链表中的遍历
		printf("%d\t",p->value);

	}
	printf("\n");

}
