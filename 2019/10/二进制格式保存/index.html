二进制格式保存

十月 2, 2019 发布在 C基础, binaryfile

学生信息保存为二进制

#include <stdio.h>
const int STR_LEN = 20;
typedef struct _student {
	char name[STR_LEN];
	int gender;
	int age;
 } Student;

void getList(Student aStu[], int number);
int save(Student aStu[], int number);

int main(int argc, char const *argv[])
{
	int number = 0;
	printf("输入学生的数量：");
	scanf("%d",&number);
	Student aStu[number];

	getList(aStu,number);
	if ( save(aStu, number) ){
		printf("保存成功\n");
	} else {
		printf("保存失败\n");
	}
	return 0;
}

void getList( Student aStu[], int number)
{
	char format[STR_LEN];//新建一个格式数组format
	sprintf(format,"%%%ds",STR_LEN-1);//产生一个格式字符串 "%19s"， %%-->%.
	int i;
	for ( i=0; i<number; i++ ){
		printf("第%d个学生：\n",i+1);
		printf("\t姓名");
		scanf(format,aStu[i].name);
		printf("\t性别(0-男，1-女，2-其他):");
		scanf("%d",&aStu[i].gender);
		printf("\t年龄：");
		scanf("%d",&aStu[i].age);
	}

}

int save(Student aStu[], int number)
{
	int ret = -1;
	FILE *fp = fopen("student.data","w");//打开student.date文件,默认输出二进制文件,与文本格式无关
	if (fp){
		ret = fwrite(aStu, sizeof(Student),number,fp);//二进制文件的读写
		fclose(fp);
	}
	return ret == number;//返回值是成功读写了几个结构体，1/0

}
