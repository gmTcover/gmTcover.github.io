int main( int argc, char const *argv[] )
{
	char s1[] = "abc";
	char s2[] = "abc";

	printf("%d\n",mycmp(s1,s2));

	printf("%d\n",'a'-'A');
	return 0;
}
int mystrcmp( const char *s1, const char *s2 )
{
	while ( *s1 == *s2 && *s1 != '\0'){
		s1++;
		s2++;
	}
	return *s1 -*s2;
}

char* mycpy( char* dst, char* src){
	int idx = 0;
	while ( src[idx]){
		dst[idx] = src[idx];
		idx++;
	}
	dst[idx] = '\0';
	return dst;
	}
